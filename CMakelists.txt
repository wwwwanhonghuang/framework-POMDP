cmake_minimum_required(VERSION 3.16)

project(pomdp_framework
    VERSION 0.1
    LANGUAGES CXX
)

# ---------------- C++ standard ----------------
set(CMAKE_CXX_STANDARD 17)
set(CMAKE_CXX_STANDARD_REQUIRED ON)
set(CMAKE_CXX_EXTENSIONS OFF)

# ---------------- Include paths ----------------
# Framework headers + example-local headers
include_directories(
    ${PROJECT_SOURCE_DIR}/include
    ${PROJECT_SOURCE_DIR}/examples
)

# ---------------- Example: offline_pf_example ----------------
add_executable(offline_pf_example
    examples/offline_pf_example/offline_main.cpp
    examples/offline_pf_example/model.cpp
)

# ---------------- Example: offline_pf_large_example ----------------
add_executable(offline_pf_large_example
    examples/offline_pf_example/offline_large_main.cpp
    examples/offline_pf_example/model.cpp
)

# ---------------- Warnings ----------------
if (MSVC)
    target_compile_options(offline_pf_example PRIVATE /W4)
    target_compile_options(offline_pf_large_example PRIVATE /W4)
else()
    target_compile_options(offline_pf_example PRIVATE -Wall -Wextra -Wpedantic)
    target_compile_options(offline_pf_large_example PRIVATE -Wall -Wextra -Wpedantic)
endif()
